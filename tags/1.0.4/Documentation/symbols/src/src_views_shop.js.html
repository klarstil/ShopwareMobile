202 		if(currIdx != 1) {
203 			var prevDepth      = currIdx - 1,
204 				prev           = this.items.getAt(prevDepth);

211 			this.syncToolbar(prev);

213 			window.setTimeout(function() {

215 			}, 250);
216 		} else {
217 			var prev = this.items.getAt(0),
218 				last = this.items.getAt(1);
219 			this.setActiveItem(prev, {
220 				type: 'slide',
221 				reverse: true,
222 				scope: this
223 			});

480     /**
481      * Handles the toolbar
482      *
483      * @param card
484      */
485 	syncToolbar: function(card) {
486 		var list          = card || this.getActiveItem(),
487 			depth         = this.items.indexOf(list),
488 			recordNode    = list.recordNode,
489 			parentNode    = recordNode ? recordNode.parentNode : null,
490 			backBtn       = this.backButton,
491 			backBtnText   = this.useTitleAsBackText && parentNode ? this.renderTitleText(parentNode) : this.backText,
492 			backToggleMth = (depth !== 0) ? 'show' : 'hide',
493 			backHomeMth   = (depth === 0) ? 'show' : 'hide',
494 			shopView      = Ext.getCmp('shop');
495 
496 			if (backBtn) {
497 				backBtn[backToggleMth]();
498 				this.backBtn[backHomeMth]();
499 				if (parentNode) {
500 					backBtn.setText(App.Helpers.truncate(backBtnText, 9));
501 				} else if(parentNode === null) {
502 					this.backBtn.show();
503 					this.toolbar.doLayout();
504 				}
505 			}
506 
507 
508 			if (this.toolbar && this.updateTitleText) {
509 				this.toolbar.setTitle(recordNode && recordNode.getRecord() ? this.renderTitleText(recordNode) : this.title || '');
510 				this.toolbar.doLayout();
511 			}
512 	}
