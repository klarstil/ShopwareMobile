 10 /**
 11  * Main cart view
 12  *
 13  * Contains all necessary cart components
 14  *
 15  * @access public
 16  * @class
 17  * @extends Ext.Panel
 18  */
 19 App.views.Cart.index = Ext.extend(Ext.Panel,
 20 /** @lends App.views.Cart.index# */
 21 {
 22 	id: 'cart',
 23 	title: 'Warenkorb',
 24 	iconCls: 'cart',
 25 	layout: 'card',
 26 	autoHeight: true,
 27 	scroll: false,
 28 	listeners: {
 29 		scope: this,
 30 		
 31 		activate: function() {
 32 			var amountEl = Ext.get('amount-display');
 33 			if(amountEl) {
 34 				amountEl.setHTML((Math.round(App.stores.Cart.amount*100) / 100) + ' €*');
 35 			}
 36 		},
 37 
 38 		deactivate: function(me) {
 39 			var active = me.getActiveItem(), view;
 40 
 41 			if(active && active.id == 'orderConfirmation') {
 42 				active.setActiveItem(me.pnl, {
 43 					type: 'slide',
 44 					reverse: true,
 45 					scope: this
 46 				});
 47 
 48 				me.toolbar.setTitle(me.title);
 49 				me.toolbar.show();
 50 
 51 				Ext.getCmp('cartlist').destroy();
 52 				active.destroy();
 53 				
 54 				if(!Ext.getCmp('cartlist')) {
 55 					view = new App.views.Cart.list;
 56 					view.tpl = App.views.Cart.emptyTpl;
 57 					Ext.getCmp('cart').add(view);
 58 					me.doComponentLayout();
 59 				}
 60 			}
 61 		}
 62 	},
 63 	
 64 	initComponent: function() {
 65 
 66 		this.checkoutBtn = new Ext.Button({
 67 			id: 'checkoutBtn',
 68 			ui: 'forward action',
 69 			text: 'Checkout',
 70 			scope: this,
 71 			handler: this.onCheckoutBtn
 72 		});
 73 
 74 		this.toolbar = new Ext.Toolbar({
 75 			dock: 'top',
 76 			title: this.title,
 77 			items: [{ xtype: 'spacer' }, this.checkoutBtn]
 78 		});
 79 
 80 		this.pnl = new Ext.Panel({
 81 			scroll: 'vertical',
 82 			height: '100%',
 83 			items: [new App.views.Cart.list]
 84 		});
 85 
 86 		Ext.apply(this, {
 87 			dockedItems: [this.toolbar],
 88 			items: [this.pnl]
 89 		});
 90 		App.views.Cart.index.superclass.initComponent.call(this);
 91 
 92 		if(App.stores.Cart.articleCount < 1) {
 93 			this.checkoutBtn.hide();
 94 		}
 95 	},
 96     
 97     /**
 98      * Will be called when the checkout btn is tapped
 99      */
100 	onCheckoutBtn: function() {
101 		var view = this;
102 
103 		Ext.dispatch({
104 			controller: 'checkout',
105 			action: 'show',
106 			parentView: view
107 		});
108 	}
109 });

111 /**
112  * Cart list
113  *
114  * Displays the cart content
115  *
116  * @access public
117  * @class
118  * @extends Ext.Panel
119  */
120 App.views.Cart.list = Ext.extend(Ext.Panel,
121 /** @lends App.views.Cart.list# */
122 {
123 	id: 'cartlist',
124 	layout: 'fit',
125 	flex: 1,
126 	tpl: App.views.Cart.indexTpl,
127 	data: App.stores.Cart,
128 	store: App.stores.Cart,
129 	autoHeight: true,
130 	scroll: false,
131 	initComponent: function() {
132 		this.store.on({
133 			datachanged: this.onDataChanged,
134 			scope: this
135 		});
136 		App.views.Cart.list.superclass.initComponent.call(this);
137 
138 		Ext.apply(this, {
139 			listeners: {
140 				el: {
141 					tap: this.onDeleteBtn,
142 					delegate: '.deleteBtn',
143 					scope: this
144 				},
145 				scope: this
146 			}
147 		})
148 	},
149 
150     /**
151      * Calls when the store is changed
152      *
153      * @param store
154      */
155 	onDataChanged: function(store) {
156 		this.update(this.store);
157 	},
158 
159     /**
160      * Calls when the store is updated
161      * @param store
162      */
163 	update: function (store) {
164 		if (store.items.length) {
165 			this.tpl = App.views.Cart.indexTpl;
166 
167 			if(this.ownerCt && this.ownerCt.ownerCt && this.ownerCt.ownerCt.checkoutBtn) {
168 				this.ownerCt.ownerCt.checkoutBtn.show();
169 			}
170 
171 			this.hideCheckoutBtn(false);
172 		} else {
173 			this.tpl = App.views.Cart.emptyTpl;
174 			
175 			if(this.ownerCt && this.ownerCt.ownerCt && this.ownerCt.ownerCt.checkoutBtn) {
176 				this.ownerCt.ownerCt.checkoutBtn.hide();
177 			}
178 			this.hideCheckoutBtn(true);
179 		}
180 		App.views.Cart.list.superclass.update.apply(this, arguments);
181 	},
182 
183     /**
184      * Calls when the delete button is tapped
185      *
186      * @param event
187      * @param el
188      */
189 	onDeleteBtn: function(event, el) {
190 		var el = Ext.get(el), val;
191 		val = el.dom.attributes[1].nodeValue;
192 		App.stores.Cart.remove(val);
193 		return false;
194 	},
195 
196     /**
197      * Hides the checkout btn
198      *
199      * @param state
200      */
201 	hideCheckoutBtn: function(state) {
202 		var btn = Ext.getCmp('checkoutBtn');
203 		if (state === true) { btn.hide(); } else { btn.show(); }
204 	}
205 });
